createAssistantV1?.listDialogNodes({
    workspaceId: WATSON_ASSISTANT_WORKSPACE
  })
    .then((listDialogNodesResponse) => {
      /// -------Search entry point node and last node------ ///
      const entryPointNode: DialogNode = newDialogNodesList?.find((node: DialogNode) => { return node?.title?.toLowerCase().includes('entrypoint') })
      const lastNode: DialogNode = listDialogNodesResponse?.result?.dialog_nodes?.find((node: any) => { return node?.title?.toLowerCase().includes('lastnode') })

      /// ------------Sets the entry point's previous_sibling with the node id of last node------ ///
      entryPointNode.previous_sibling = lastNode?.dialog_node

      createAssistantV1.updateWorkspace({
        workspaceId: WATSON_ASSISTANT_WORKSPACE,
        dialogNodes: listDialogNodesResponse.result.dialog_nodes.concat(newDialogNodesList)
      })
        .then(updateWorkspaceResponse => {
          console.log('updateWorkspaceResponse: ', updateWorkspaceResponse)
        })
        .catch(updateWorkspaceError => {
          console.log('updateWorkspaceError: ', updateWorkspaceError)
        })
    })
    .catch(listDialogNodesError => {
      console.log('listDialogNodesError: ', listDialogNodesError)
    })
